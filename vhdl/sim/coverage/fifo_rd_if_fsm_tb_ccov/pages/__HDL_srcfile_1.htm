     14  use     ieee.numeric_std.all    ;

     20  ---------------------------------------------------------------------------
     21  entity fifo_rd_if_fsm is
     22      generic(
     23          DW          :       integer :=  32  -- Data width
     24      );
     25      port(
     26          clk         :   in  std_logic;
     27          rstn        :   in  std_logic;
     28          wr          :   in  std_logic;
     29          -- coverage off t
     30          wdata       :   in  std_logic_vector(DW-1 downto 0);
     31          -- coverage on
     32          fifo_rd     :   in  std_logic;
     33          trig_in_fsm :   in  std_logic;
     34          wdata_reg_en:   out std_logic;
     35          empty       :   out std_logic;
     36          got_read_out:   out std_logic;
     37          enabled_fsm :   out std_logic
     38      );
     39  end entity fifo_rd_if_fsm;
     40  ---------------------------------------------------------------------------
     41  
     42  architecture    rtl of fifo_rd_if_fsm   is
     43  
     44      type    state_t is(
     45          IDLE                ,
     46          ENABLED             ,
     47          RDATA_PRESENT       ,
     48          RDATA_GOT_READ_OUT
     49      );
     50      signal  cur_state   :   state_t;
     51      signal  nxt_state   :   state_t;
     52  
     53  begin
     54  
     55      L_FSM:  block
     56      begin
     57  
     58          ----------------------------------------------------------------------
     59          L_NEXT_STATE:   process(cur_state,wr,wdata,fifo_rd,trig_in_fsm) is
     60          begin
     61  
     62              nxt_state   <= cur_state;
     63  
     64              case(cur_state) is
     65                  -----------------------------------------------------
     66                  -- Another component can also enable the module via 'trig_in_fsm'
     67                  when IDLE           =>      if( (wdata(1 downto 0) = b"01" and wr = '1') or
     68                                                  (trig_in_fsm = '1')
     69                                              ) then
     70                                                  nxt_state   <= ENABLED;
     71                                              end if;
     72                  -----------------------------------------------------
     73                  when ENABLED        =>      --if( wdata(1 downto 0) = b"10"    and wr = '1')  then
     74                                              --    nxt_state   <= IDLE;
     75                                              --elsif( wr = '1') then
     76                                              if( wr = '1') then
     77                                                  nxt_state   <= RDATA_PRESENT;
     78                                              end if;
     79                  -----------------------------------------------------
     80                  -- Clearing has prioprity
     81                  when RDATA_PRESENT  =>      if( wdata(1 downto 0) = b"10"    and wr = '1')  then
     82                                                  nxt_state   <= IDLE;
     83                                              elsif(fifo_rd = '1')  then
     84                                                  nxt_state   <= RDATA_GOT_READ_OUT;
     85                                              end if;
     86                  -----------------------------------------------------
     87                  -- Clearing has prioprity
     88                  -- Could be useful to be able to send back to enabled immediately
     89                  when RDATA_GOT_READ_OUT =>  if(wdata(1 downto 0) = b"10"    and wr = '1') then
     90                                                  nxt_state   <= IDLE;
     91                                              elsif(  (wdata(1 downto 0) = b"01" and wr = '1') or
     92                                                      (trig_in_fsm = '1')
     93                                              ) then
     94                                                  nxt_state   <= ENABLED;
     95                                              end if;
     96                  -----------------------------------------------------
     97                  -- coverage off
     98                  when others =>  nxt_state   <= IDLE;
     99                  -- coverage on
     100                  -----------------------------------------------------
     101              end case;
     102          end process;
     103          ----------------------------------------------------------------------
     104  
     105          -- FSM outputs
     106  
     107          wdata_reg_en    <=  '1' when( cur_state = ENABLED
     108                                      ) else '0';
     109          -------------------------------------------------------------
     110          empty           <=  '1' when( cur_state = IDLE      or
     111                                        cur_state = ENABLED   or
     112                                        cur_state = RDATA_GOT_READ_OUT
     113                                      ) else '0';
     114          -------------------------------------------------------------
     115          got_read_out    <=  '1' when( cur_state = RDATA_GOT_READ_OUT
     116                                      ) else '0';
     117          -------------------------------------------------------------
     118          enabled_fsm     <=  '1' when( cur_state /= IDLE
     119                                      ) else '0';
     120  
     121  
     122          -----------------------------------------------------
     123          L_STATE:    process(clk,rstn)   is
     124          begin
     125              if(rstn = '0')  then
     126                  cur_state  <= IDLE;
     127  
     128              elsif(rising_edge(clk)) then
     129                  cur_state  <= nxt_state;
     130  
     131              end if;
     132          end process;
     133          -----------------------------------------------------
     134      end block;
     135  
     136  end architecture;
