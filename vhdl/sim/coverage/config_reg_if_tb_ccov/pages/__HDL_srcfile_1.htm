     12  library ieee                    ;
     13  use     ieee.std_logic_1164.all ;
     14  use     ieee.numeric_std.all    ;

     17  
     18  
     19  ---------------------------------------------------------------------------
     20  entity config_reg_if is
     21      generic(
     22          ACK_NEEDED      :       boolean :=  false   ;
     23          REG_LAYER       :       boolean :=  false   ;
     24          DW              :       integer :=  32      ;
     25          AW              :       integer :=  32      ;
     26          RD_START        :       std_logic_vector(63 downto 0):=     X"00000000FFFF0000"; -- WorkAround: Can't use 'DW' here, so made it wide enough
     27          WR_START        :       std_logic_vector(63 downto 0):=     X"000000000000FFFF"  --
     28      );
     29      port(
     30          clk             :   in  std_logic;
     31          rstn            :   in  std_logic;
     32          wr              :   in  std_logic                           ;--
     33          -- coverage off t
     34          wdata           :   in  std_logic_vector(DW-1   downto 0)   ;--
     35          -- coverage on
     36          rd              :   in  std_logic                           ;-- MMIO Slave IF
     37          -- coverage off t
     38          rdata           :   out std_logic_vector(DW-1   downto 0)   ;--
     39          -- coverage on
     40          wstrb_to_DUV    :   out std_logic                           ;--
     41          rstrb_to_DUV    :   out std_logic                           ;--
     42          -- coverage off t
     43          wdata_to_DUV    :   out std_logic_vector(DW-1   downto 0)   ;-- Simple config register IF
     44          addr_to_DUV     :   out std_logic_vector(AW-1   downto 0)   ;--
     45          rdata_from_DUV  :   in  std_logic_vector(DW-1   downto 0)   ;--
     46          -- coverage on
     47          cs_to_DUV       :   out std_logic                           ;-- Chip Select for the config register
     48          ack_from_DUV    :   in  std_logic                            -- Ack. input, when ACK_NEEDED is set to TRUE
     49      );
     50  end entity config_reg_if;
     51  ---------------------------------------------------------------------------
     52  
     53  architecture rtl of config_reg_if is
     54  
     55      signal  fsm_addr_en :   std_logic;
     56      signal  fsm_wdata_en:   std_logic;
     57      signal  fsm_rdata_en:   std_logic;
     58  
     59      signal  fsm_rstrb   :   std_logic;
     60      signal  fsm_wstrb   :   std_logic;
     61      signal  fsm_cs      :   std_logic;
     62  
     63      -- coverage off t
     64      signal  rdata_reg   :   std_logic_vector(DW-1 downto 0);
     65      signal  wdata_reg   :   std_logic_vector(DW-1 downto 0);
     66      signal  addr_reg    :   std_logic_vector(AW-1 downto 0);
     67      -- coverage on 
     68  
     69  
     70      signal  cs_reg      :   std_logic;
     71      signal  wstrb_reg   :   std_logic;
     72      signal  rstrb_reg   :   std_logic;
     73  
     74  
     75  begin
     76  
     77      L_DUV_IF:   block
     78                  begin
     79                      addr_to_DUV     <=  addr_reg;
     80                      wdata_to_DUV    <=  wdata_reg;
     81                  end block;
     82  
     83  
     84      -----------------------------------------------------
     85      L_RD_IF: block
     86      begin
     87          -------------------------------------------------
     88          process(clk) is
     89          begin
     90              if(rising_edge(clk))    then
     91  
     92                  if(fsm_rdata_en = '1')   then
     93                      rdata_reg   <= rdata_from_DUV;
     94                  end if;
     95  
     96              end if;
     97          end process;
     98  
     99          -------------------------------------------------
     100          rdata <=    rdata_reg       when(rd = '1')   else
     101                      (others => '0');
     102          -------------------------------------------------
     103      end block;
     104      -----------------------------------------------------
     105  
     106  
     107      -----------------------------------------------------
     108      L_WR_IF: block
     109      begin
     110          -------------------------------------------------
     111          process(clk) is
     112          begin
     113              if(rising_edge(clk))    then
     114  
     115                  if(fsm_wdata_en = '1')   then
     116                      wdata_reg   <= wdata;
     117                  end if;
     118  
     119              end if;
     120          end process;
     121          -------------------------------------------------
     122          process(clk) is
     123          begin
     124              if(rising_edge(clk))    then
     125  
     126                  if(fsm_addr_en = '1')    then
     127                      addr_reg   <= std_logic_vector(resize(signed(wdata), AW)); -- Resizing to address width
     128                  end if;
     129  
     130              end if;
     131          end process;
     132          -------------------------------------------------
     133  
     134      end block;
     135      -----------------------------------------------------
     136  
     137      -----------------------------------------------------
     138      L_REG_LAYER: if(REG_LAYER = true)    generate
     139          -------------------------------------------------
     140          process(clk,rstn) is
     141          begin
     142              if(rstn = '0')  then
     143                  cs_reg  <= '0';
     144  
     145              elsif(rising_edge(clk))    then
     146                  cs_reg  <= fsm_cs;
     147  
     148              end if;
     149          end process;
     150          -------------------------------------------------
     151          cs_to_DUV  <= cs_reg;
     152          -------------------------------------------------
     153          process(clk,rstn) is
     154          begin
     155              if(rstn = '0')  then
     156                  wstrb_reg   <= '0';
     157  
     158              elsif(rising_edge(clk))    then
     159                  wstrb_reg  <= fsm_wstrb;
     160  
     161              end if;
     162          end process;
     163          -------------------------------------------------
     164          wstrb_to_DUV    <= wstrb_reg;
     165          -------------------------------------------------
     166          process(clk,rstn) is
     167          begin
     168              if(rstn = '0')  then
     169                  rstrb_reg   <= '0';
     170  
     171              elsif(rising_edge(clk))    then
     172                  rstrb_reg  <= fsm_rstrb;
     173  
     174              end if;
     175          end process;
     176          -------------------------------------------------
     177          rstrb_to_DUV    <= rstrb_reg;
     178          -------------------------------------------------
     179      end generate;
     180      -----------------------------------------------------
     181  
     182      -----------------------------------------------------
     183      L_NO_REG_LAYER: if(REG_LAYER = false)   generate
     184          cs_to_DUV       <= fsm_cs   ;
     185          wstrb_to_DUV    <= fsm_wstrb;
     186          rstrb_to_DUV    <= fsm_rstrb;
     187      end generate;
     188      -----------------------------------------------------
     189  
     190  
     191      -----------------------------------------------------
     192      -- 	Instantiating the 'config_reg_if_fsm' FSM.
     193      L_FSM:
     194          entity work.config_reg_if_fsm(rtl)
     195              generic map(
     196                  ACK_NEEDED      =>  ACK_NEEDED  ,
     197                  DW              =>  DW          ,
     198                  RD_START        =>  RD_START    ,
     199                  WR_START        =>  WR_START
     200              )
     201              port map(
     202                  clk             => clk          ,
     203                  rstn            => rstn         ,
     204                  wr              => wr           ,
     205                  wdata           => wdata        ,
     206                  ack_from_DUV    => ack_from_DUV ,
     207                  rstrb           => fsm_rstrb    ,
     208                  wstrb           => fsm_wstrb    ,
     209                  cs              => fsm_cs       ,
     210                  addr_en         => fsm_addr_en  ,
     211                  wdata_en        => fsm_wdata_en ,
     212                  rdata_en        => fsm_rdata_en
     213              );
     214      -----------------------------------------------------
     215  
     216  end architecture rtl;
