     39  -----------------------------------------------------------------------------------------------
     40  entity sc_ff_tb is
     41      -- These generics are initialized by the Python script at elaboration time.
     42      generic(
     43  
     44          constant    RTL_NAME_G  :           string;         -- RTL name
     45  
     46          constant    CLI_G       :           integer := 0;   --
     47          constant    GUI_G       :           integer := 0;   -- Simulation type indicators.
     48          constant    RGR_G       :           integer := 0    --
     49      );
     50  end entity;
     51  -----------------------------------------------------------------------------------------------
     52  
     53  
     54  architecture bhv of sc_ff_tb is
     55  
     56      constant    clk_enabled_c:  std_logic:= '1';            -- Clock is enabled by default.
     57  
     58  
     59  
     60      --------- Initializing the generic parameters  ----------
     61      procedure run_test is new tc
     62          generic map(
     63              new_run_name        =>  "run_test"      ,       -- constant string
     64  
     65              rtl_in_if_t         =>  sc_ff_in_if_t     ,       -- type
     66              called_tc           =>  sc_ff_test        ,       -- procedure
     67              called_tc_name      =>  "sc_ff_test"              -- constant string
     68          );
     69      -----------------------------------------------------------------------------------
     70  
     71      --------- Initializing the generic parameters  ----------
     72      procedure run_check is new chk
     73          generic map(
     74              new_run_name        =>  "run_check"     ,       -- constant string
     75  
     76              rtl_out_if_t        =>  sc_ff_out_if_t    ,       -- type
     77              called_chk          =>  sc_ff_check       ,       -- procedure
     78              called_chk_name     =>  "sc_ff_check"             -- constant string
     79          );
     80      -----------------------------------------------------------------------------------
     81          -- Shared variable between 'tc' and 'chk' process
     82      shared variable sync_sv :   synchronizer_t;
     83  
     84  
     85      signal  rtl_out_if      :   sc_ff_out_if_t;
     86  
     87      signal      tb_if       :   tb_if_t :=( --
     88          clk     =>  '1',                    --
     89          clk_en  =>  '0',                    -- Testbench resouce related signals.
     90          rstn    =>  '1',                    --
     91          rstn_req=>  '0'                     --
     92      );                                      --
     93  
     94  begin
     95  
     96  
     97      -----------------------------------------------------------------------------------------
     98      tc:     process is
     99                  constant    this                :   string      :=  "process_tc";
     100  
     101                  variable    clock_enable_v      :   std_logic   :=  clk_enabled_c;          -- Clock enable, enabled by default
     102              begin
     103                  ------------------------------------------------------
     104                  setup(RTL_NAME_G,this,              clock_enable_v  ,                       --
     105                                                      tb_if.clk_en                            -- Setup the testbench.
     106                                                                      );                      --
     107                  ------------------------------------------------------
     108                  run_test(RTL_NAME_G,this,sync_sv,   rtl_in_if           ,                   --
     109                                                      tb_if.clk           ,
     110                                                      tb_if.rstn_req      ,
     111                                                      check_descriptor    );-- Run the testcases for the chosen RTL
     112                  -- Drain time
     113                  wait for 100 ns;
     114                  ------------------------------------------------------
     115                  report_and_exit(RTL_NAME_G,this,    RGR_G           ,
     116                                                      check_descriptor);                      -- Generate report based on results and exit.
     117                  ------------------------------------------------------
     118              end process;
     119      -----------------------------------------------------------------------------------------
     120  
     121  
     122      -----------------------------------------------------------------------------------------
     123      chk:    process is
     124                  constant    this            :   string      :=  "process_chk";
     125              begin
     126                  ------------------------------------------------------
     127                  run_check(RTL_NAME_G,this);
     128                  ------------------------------------------------------
     129              end process;
     130      -----------------------------------------------------------------------------------------
     131  
     132  
     133  
     134  
     135  
     136      -----------------------------------------------------------------------------------------
     137      rst_gen:    process is
     138                      constant    this            :   string      :=  "process_rst_gen";
     139                  begin
     140  
     141                      wait on tb_if.rstn_req;
     142  
     143                      if(tb_if.rstn_req = '1')    then
     144                          tb_if.rstn     <=  '0'; print(this &": Reset is asserted.", 1);
     145                      else
     146                          tb_if.rstn     <=  '1'; print(this &": Reset is de-asserted.", 1);
     147                      end if;
     148  
     149                  end process;
     150      -----------------------------------------------------------------------------------------
     151  
     152  
     153  
     154      -----------------------------------------------------------------------------------------
     155      clk_gen:    tb_if.clk  <=  not tb_if.clk  after (clk_per_c/2) when (tb_if.clk_en = '1') else '1';
     156      -----------------------------------------------------------------------------------------
     157      L_DUT:  entity work.sc_ff(rtl)
     158                  port map(
     159                      clk         =>  tb_if.clk       ,
     160                      rstn        =>  tb_if.rstn      ,
     161                      set         =>  rtl_in_if.set   ,
     162                      clr         =>  rtl_in_if.clr   ,
     163                      q           =>  rtl_out_if.q
     164  
     165                  );
     166      -----------------------------------------------------------------------------------------
     167  
     168  end architecture;
