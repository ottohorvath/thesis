     12  library ieee                    ;
     13  use     ieee.std_logic_1164.all ;
     14  use     ieee.numeric_std.all    ;
     15  
     16  ---------------------------------------------------------------------------
     17  entity chg_det is
     18  
     19      port(
     20          clk             :   in  std_logic;
     21          rstn            :   in  std_logic;
     22          en              :   in  std_logic;
     23          clr             :   in  std_logic;
     24          sig             :   in  std_logic;
     25          det_out         :   out std_logic
     26      );
     27  end entity chg_det;
     28  ---------------------------------------------------------------------------
     29  
     30  
     31  ---------------------------------------------------------------------------
     32  architecture rtl of chg_det is
     33  
     34      signal  sample      :   std_logic;
     35      signal  en_sampling :   std_logic;
     36  
     37  begin
     38  
     39      ---------------------------------------------------------------
     40      L_SAMPLER:  process(clk,rstn)  is
     41      begin
     42          if(rstn = '0')  then
     43              sample <= '0';
     44  
     45          elsif(rising_edge(clk)) then
     46              -----------------------------
     47              if(en_sampling = '1')   then
     48                  sample  <= sig;
     49  
     50              end if;
     51              -----------------------------
     52          end if;
     53      end process;
     54      ---------------------------------------------------------------
     55      L_FSM:      entity work.chg_det_fsm(rtl)
     56                      port map(
     57                          clk             =>  clk         ,
     58                          rstn            =>  rstn        ,
     59                          en_fsm          =>  en          ,
     60                          clr_fsm         =>  clr         ,
     61                          sig_fsm         =>  sig         ,
     62                          sample_fsm      =>  sample      ,
     63                          en_sampling_fsm =>  en_sampling ,
     64                          sig_changed_fsm =>  det_out
     65                      );
     66      ---------------------------------------------------------------------
     67  end architecture rtl;
