     16  library ieee                    ;
     17  use     ieee.std_logic_1164.all ;
     18  use     ieee.numeric_std.all    ;
     19  
     20  
     21  ---------------------------------------------------------------------------
     22  entity fifo_rd_if is
     23      generic (
     24          REG_LAYER       :           boolean :=  false;  -- For timing closure purposes
     25          DW              :           integer :=  32      -- Data width
     26      );
     27      port (
     28          clk             :       in  std_logic                           ;
     29          rstn            :       in  std_logic                           ;-- Async. active LOW reset
     30          trig_in         :       in  std_logic                           ;--
     31          trig_out_0      :       out std_logic                           ;-- Trigger IF
     32          trig_out_1      :       out std_logic                           ;
     33          rdata           :       out std_logic_vector(DW-1 downto 0) ;
     34          wr              :       in  std_logic                           ;
     35          wdata           :       in  std_logic_vector(DW-1  downto 0);
     36          empty_to_DUV    :       out std_logic                           ;--
     37          rd_from_DUV     :       in  std_logic                           ;-- Read side
     38          rdata_to_DUV    :       out std_logic_vector(DW-1 downto 0)      --
     39      );
     40  end entity;
     41  ---------------------------------------------------------------------------------------------------
     42  architecture rtl of fifo_rd_if is
     43  
     44      --------------------------------------------------------------------------------------------------
     45      function slv(
     46          N: natural;
     47          B: std_logic
     48      )   return std_logic_vector  -- Replacate std_logic-s to slv, like Verilog's replicate operator (e.g. {32{1b1}} )
     49      is
     50          variable result: std_logic_vector(1 to N);
     51      begin
     52          for i in 1 to N loop
     53              result(i) := B;
     54          end loop;
     55  
     56          return result;
     57      end;
     58      --------------------------------------------------------------------------------------------------
     59      --------------------------------------------------------------------------------------------------
     60      --------------------------------------------------------------------------------------------------
     61  
     62  
     63  
     64      -- Read Data output
     65      -- ================
     66      -- [0]: Shows if the module is enabled
     67      -- [1]: Shows the value of 'empty_to_DUV'
     68      -- [2]: Shows if data got read out
     69      signal      rdata_int_w     :   std_logic_vector(DW-1 downto 0);
     70  
     71  
     72      -- Write Data Register
     73      -- ===================
     74      -- Stores write data which will be eventually read out by the DUV
     75      signal      wdata_reg       :   std_logic_vector(DW-1 downto 0);
     76      signal      wdata_reg_en    :   std_logic;
     77  
     78  
     79      -- Signals from FSM
     80      -- ================
     81      signal      data_got_read_out   :   std_logic;
     82      signal      fifo_empty          :   std_logic;
     83      signal      enabled_from_fsm    :   std_logic;
     84  
     85  
     86      -- Register layer on output
     87      -- ========================
     88      signal      fifo_empty_reg      :   std_logic;
     89  
     90  
     91      -- Trigger output registers
     92      -- ========================
     93      signal      trig_out_0_reg      :   std_logic;
     94      signal      trig_out_1_reg      :   std_logic;
     95  
     96  begin
     97  
     98      -------------------------------------------------------------
     99      L_WDATA_REG:    process(clk,rstn) is
     100                      begin
     101                          if(rstn = '0')  then
     102                              wdata_reg   <= slv(DW, '0');
     103  
     104                          elsif( rising_edge(clk))    then
     105  
     106                              if(wdata_reg_en = '1')    then
     107                                  wdata_reg   <= wdata;
     108                              end if;
     109  
     110                          end if;
     111                      end process;
     112      -------------------------------------------------------------
     113      L_RDATA:    block
     114                  begin
     115                      rdata   <=  slv(DW-3, '0')  &
     116                                  data_got_read_out   &   -- [2]
     117                                  fifo_empty          &   -- [1]
     118                                  enabled_from_fsm    ;   -- [0]
     119                  end block;
     120      -------------------------------------------------------------
     121      L_DUV_SIDE: block
     122                  begin
     123                      ---------------------------------------------
     124                      L_EMPTY_REG:    if(REG_LAYER = true)    generate
     125                                          -------------------------------------
     126                                          process(clk,rstn) is
     127                                          begin
     128                                              if(rstn = '0')    then
     129                                                  fifo_empty_reg  <= '1';
     130  
     131                                              elsif(rising_edge(clk)) then
     132                                                  fifo_empty_reg  <= fifo_empty;
     133  
     134                                              end if;
     135                                          end process;
     136                                          -------------------------------------
     137                                          process(clk,rstn) is
     138                                          begin
     139                                              if(rstn = '0')    then
     140                                                  trig_out_0_reg  <= '0';
     141  
     142                                              elsif(rising_edge(clk)) then
     143                                                  trig_out_0_reg  <= enabled_from_fsm;
     144  
     145                                              end if;
     146                                          end process;
     147                                          -------------------------------------
     148                                          process(clk,rstn) is
     149                                          begin
     150                                              if(rstn = '0')    then
     151                                                  trig_out_1_reg  <= '0';
     152  
     153                                              elsif(rising_edge(clk)) then
     154                                                  trig_out_1_reg  <= data_got_read_out;
     155  
     156                                              end if;
     157                                          end process;
     158                                          -------------------------------------
     159  
     160                                          -- Drive the output from register
     161                                          empty_to_DUV    <=  fifo_empty_reg;
     162                                          trig_out_0      <=  trig_out_0_reg;
     163                                          trig_out_1      <=  trig_out_1_reg;
     164  
     165  
     166                                      end generate;
     167                      ---------------------------------------------
     168                      L_NO_REG_LAYER: if(REG_LAYER = false)   generate
     169  
     170                                          -- Drive output signal from FSM
     171                                          empty_to_DUV    <=  fifo_empty;
     172                                          trig_out_0      <=  enabled_from_fsm;
     173                                          trig_out_1      <=  data_got_read_out;
     174  
     175                                      end generate;
     176                      ---------------------------------------------
     177  
     178                      -- Drive the data from processor to the DUV IF
     179                      rdata_to_DUV    <= wdata_reg;
     180                      ---------------------------------------------
     181                  end block;
     182      -------------------------------------------------------------
     183  
     184  
     185      -------------------------------------------------------------
     186      L_FSM:  entity  work.fifo_rd_if_fsm(rtl)
     187                  generic map(
     188                      DW              =>  DW
     189                  )
     190                  port map(
     191                      clk             =>  clk                 ,
     192                      rstn            =>  rstn                ,
     193                      trig_in_fsm     =>  trig_in             ,
     194                      wr              =>  wr                  ,
     195                      wdata           =>  wdata               ,
     196                      fifo_rd         =>  rd_from_DUV         ,
     197                      wdata_reg_en    =>  wdata_reg_en        ,
     198                      empty           =>  fifo_empty          ,
     199                      got_read_out    =>  data_got_read_out   ,
     200                      enabled_fsm     =>  enabled_from_fsm
     201                  );
     202      -------------------------------------------------------------
     203  
     204  
     205  
     206  
     207  
     208  end architecture;
