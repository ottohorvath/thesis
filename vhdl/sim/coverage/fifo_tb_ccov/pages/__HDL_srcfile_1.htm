     13  library ieee                    ;
     14  use     ieee.std_logic_1164.all ;
     15  use     ieee.numeric_std.all    ;

     17

     19  ---------------------------------------------------------------------------
     20  entity fifo is
     21      generic (
     22          DEPTH       :       integer range 8 to 512  :=  8           ;-- Depth of the FIFO
     23          DW          :       integer                 :=  32          -- Data width
     24      );
     25      port (
     26          clk         :       in  std_logic                           ;
     27          rstn        :       in  std_logic                           ;-- Async. active LOW reset
     28  
     29          empty       :       out std_logic                           ;--
     30          rd          :       in  std_logic                           ;-- Read side
     31          -- coverage off t
     32          rdata       :       out std_logic_vector(DW-1 downto 0) ;--
     33          -- coverage on
     34  
     35          full        :       out std_logic                           ;--
     36          wr          :       in  std_logic                           ;-- Write side
     37          -- coverage off t
     38          wdata       :       in  std_logic_vector(DW-1 downto 0)  --
     39          -- coverage on
     40      );
     41  end entity;
     42  ---------------------------------------------------------------------------------------------------
     43  architecture rtl of fifo is
     44      -----------------------------------------------------------------
     45      function clog2  (
     46          arg     :   in  natural
     47      )   return          natural
     48      is
     49          variable nbits : integer;
     50          variable num   : integer;
     51      begin
     52          num   := arg;
     53          nbits := 0;
     54  
     55          while num > 1 loop
     56  
     57              nbits   := nbits +1;
     58              num     := num / 2;
     59  
     60          end loop;
     61  
     62          return nbits;
     63      end function;
     64      -----------------------------------------------------------------
     65  
     66  
     67      function and_reduc(
     68          slv :   in std_logic_vector
     69      ) return std_logic
     70      is
     71          variable res_v : std_logic:= '1';
     72      begin
     73          for i in slv'range loop
     74              res_v := res_v and slv(i);
     75          end loop;
     76  
     77          return res_v;
     78      end function;
     79      
     80      
     81      
     82      
     83      
     84      
     85      
     86      
     87      
     88      
     89      
     90      type data_t is  array (0 to DEPTH-1) of std_logic_vector(DW-1 downto 0);
     91  
     92      -- FIFO slots
     93      -- ==========
     94      signal  payload : data_t;
     95  
     96  
     97      -- FIFO occupancy counter, wider with 1 bit than clog(DEPTH)
     98      -- because its MSB will be used for indicating full flag
     99      -- ======================================================
     100      signal  occupancy   :   std_logic_vector(clog2(DEPTH) downto 0);
     101  
     102      -- Address pointer, it is 1 bit wider than clog(DEPTH), too
     103      -- because it will be used for empty flag generation
     104      -- ======================================================
     105      signal  addr        :   std_logic_vector(clog2(DEPTH) downto 0);
     106  
     107  
     108  
     109  begin
     110      ------------------------------------------------------------------
     111      L_SHR:      process(clk)  is
     112                  begin
     113                      if( rising_edge(clk) ) then
     114                          if(wr = '1')    then
     115                              for i in 1 to DEPTH-1 loop
     116                                  payload(i)  <= payload(i-1);
     117                              end loop;
     118                              payload(0)  <=  wdata;
     119                          end if;
     120                      end if;
     121                  end process;
     122      ------------------------------------------------------------------
     123      L_OCCUP:    process(clk, rstn)  is
     124                  begin
     125                      if(rstn = '0')  then
     126                          occupancy   <= (others =>'0');
     127                      elsif( rising_edge(clk) ) then
     128                          -- Increment occupancy on write
     129                          if(wr = '1' and rd = '0')    then
     130                              occupancy   <=  std_logic_vector(unsigned(occupancy) + 1);
     131                          end if;
     132                          -- Decrement on read
     133                          if(wr = '0' and rd = '1')    then
     134                              occupancy   <=  std_logic_vector(unsigned(occupancy) - 1);
     135                          end if;
     136                      end if;
     137                  end process;
     138      ------------------------------------------------------------------
     139      L_ADDR:     process(clk, rstn)  is
     140                  begin
     141                      if(rstn = '0')  then
     142                          addr   <= (others =>'1');
     143                      elsif( rising_edge(clk) ) then
     144                          -- Increment occupancy on write
     145                          if(wr = '1' and rd = '0')    then
     146                              addr   <=  std_logic_vector(unsigned(addr) + 1);
     147                          end if;
     148                          -- Decrement on read
     149                          if(wr = '0' and rd = '1')    then
     150                              addr   <=  std_logic_vector(unsigned(addr) - 1);
     151                          end if;
     152                      end if;
     153                  end process;
     154      ------------------------------------------------------------------
     155      L_OUTPUTS:  block
     156                  begin
     157                      empty   <=  addr(clog2(DEPTH));
     158                      full    <=  occupancy(clog2(DEPTH));
     159                      rdata   <=  payload(    to_integer(unsigned(addr(clog2(DEPTH)-1 downto 0))) );
     160                  end block;
     161  
     162  
     163  end architecture;
