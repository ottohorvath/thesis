     12  library ieee                    ;
     13  use     ieee.std_logic_1164.all ;
     14  use     ieee.numeric_std.all    ;

     16  
     17  
     18  
     19  ---------------------------------------------------------------------------
     20  entity fe_det is
     21      generic(
     22          REG_LAYER       :       boolean     :=  false  -- For timing closure purposes
     23      );
     24      port(
     25          clk             :   in  std_logic;
     26          rstn            :   in  std_logic;
     27          wr              :   in  std_logic;
     28          wdata           :   in  std_logic_vector(1  downto  0);
     29          rdata           :   out std_logic;
     30          trig_out        :   out std_logic;
     31          signal_from_DUV :   in  std_logic
     32      );
     33  end entity fe_det;
     34  ---------------------------------------------------------------------------
     35  
     36  
     37  ---------------------------------------------------------------------------
     38  architecture rtl of fe_det is
     39  
     40  
     41      -- Write Data Register
     42      -- ===================
     43      -- [0]: Module is enabled
     44      -- [1]: Module is cleared
     45      -- These bits are fed into the FSM as inputs
     46      signal      wdata_reg   :   std_logic_vector(1 downto 0);
     47  
     48  
     49      -- Falling-edge detector signals
     50      -- =============================
     51      signal      fe_det_reg      :   std_logic;
     52      signal      fe_det_reg_en   :   std_logic;
     53      signal      fe_det_out      :   std_logic;
     54  
     55  
     56      -- Signals from FSM
     57      -- ===================
     58      signal      fsm_fe_caught       :   std_logic;
     59  
     60  
     61      -- Trigger output signals
     62      -- ======================
     63      signal      trig_out_reg        :   std_logic;
     64  
     65  begin
     66      ---------------------------------------------------------------------
     67      L_WDATA:    block
     68                  begin
     69                      -----------------------------------------------------
     70                      L_WDATA_REG:    process(clk,rstn) is
     71                                      begin
     72                                          if(rstn = '0') then
     73                                              wdata_reg <= b"00";
     74  
     75                                          elsif(rising_edge(clk)) then
     76  
     77                                              if(wr = '1')    then
     78                                                  wdata_reg   <= wdata;
     79                                              end if;
     80  
     81                                          end if;
     82                                      end process;
     83                      -----------------------------------------------------
     84                  end block;
     85      ---------------------------------------------------------------------
     86      L_TRIG:     block
     87                  begin
     88                      L_REG:  if(REG_LAYER = true)    generate
     89                                  process(clk,rstn) is
     90                                  begin
     91                                      if(rstn = '0')  then
     92                                          trig_out_reg    <= '0';
     93                                      elsif(rising_edge(clk)) then
     94                                          trig_out_reg    <= fsm_fe_caught;
     95                                      end if;
     96                                  end process;
     97  
     98                                  -- Drive trig_out from flop
     99                                  trig_out    <= trig_out_reg;
     100                              end generate;
     101  
     102                      L_NOREG:if(REG_LAYER = false)   generate
     103  
     104                                  -- Drive it directly
     105                                  trig_out    <= fsm_fe_caught;
     106                              end generate;
     107                  end block;
     108      ---------------------------------------------------------------------
     109      L_RDATA:    block
     110                  begin
     111                      -- FSM caught the falling-edge event
     112                      rdata   <=  fsm_fe_caught;
     113                  end block;
     114      ---------------------------------------------------------------------
     115      L_FE_DETECTOR:  block
     116                      begin
     117                          -----------------------------------------------------
     118                          L_FE_DET_REG:   process(clk,rstn) is
     119                                          begin
     120                                              if(rstn = '0')  then
     121                                                  fe_det_reg <= '0';
     122  
     123                                              elsif(rising_edge(clk)) then
     124  
     125                                                  -- Only sampling when it is needed
     126                                                  if(fe_det_reg_en = '1') then
     127                                                      fe_det_reg  <= signal_from_DUV;
     128                                                  end if;
     129  
     130                                              end if;
     131                                          end process;
     132                          -----------------------------------------------------
     133                          -- Enablement is coming from 'wdata_reg'
     134                          fe_det_reg_en   <= wdata_reg(0);
     135                          -----------------------------------------------------
     136                          -- Detector output
     137                          fe_det_out      <=  fe_det_reg  and not(signal_from_DUV);
     138                      end block;
     139      ---------------------------------------------------------------------
     140      L_FSM:  entity work.fe_det_fsm(rtl)
     141                  port map(
     142                      clk             =>  clk             ,
     143                      rstn            =>  rstn            ,
     144                      en              =>  wdata_reg(0)    ,
     145                      clr             =>  wdata_reg(1)    ,
     146                      sig_from_fe_det =>  fe_det_out      ,
     147                      fe_caught       =>  fsm_fe_caught
     148                  );
     149      ---------------------------------------------------------------------
     150  
     151  end architecture rtl;
