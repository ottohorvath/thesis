     12  library ieee                    ;
     13  use     ieee.std_logic_1164.all ;
     14  use     ieee.numeric_std.all    ;
     15  
     16  
     17  ---------------------------------------------------------------------------
     18  entity evnt_cntr is
     19      generic(
     20          SIG_W           :   integer:=   1;       -- CW of the input signal which is watched
     21          REG_LAYER       :   boolean:=   false;
     22          CW              :   integer:=   7
     23      );
     24      port(
     25          clk             :   in  std_logic;
     26          rstn            :   in  std_logic;
     27          wr              :   in  std_logic;
     28          wdata           :   in  std_logic_vector(1  downto  0);
     29          -- coverage off t
     30          rdata           :   out std_logic_vector(CW-1   downto  0);
     31          -- coverage on
     32          signal_from_DUV :   in  std_logic_vector(SIG_W-1 downto 0);
     33          trig_in         :   in  std_logic;
     34          trig_out        :   out std_logic
     35      );
     36  end entity evnt_cntr;
     37  ---------------------------------------------------------------------------
     38  
     39  
     40  ---------------------------------------------------------------------------
     41  architecture rtl of evnt_cntr is
     42  
     43      -- Write Data Register
     44      -- ===================
     45      signal  wdata_reg   :   std_logic_vector(1 downto 0);
     46  
     47  
     48      -- Signals from/to 'cntr'
     49      -- ===================
     50      -- coverage off t
     51      signal  cntr_out_w  :   std_logic_vector(CW-1 downto 0);
     52      -- coverage on
     53      signal  cntr_en_w   :   std_logic;
     54  
     55  
     56      -- Register layer
     57      -- ==============
     58      signal  trig_out_reg    :   std_logic;
     59      signal  trig_out_reg_en :   std_logic;
     60  
     61  
     62      -- SC FF on 'trig_in' input
     63      -- ========================
     64      signal  trig_in_sc_ff   :   std_logic;
     65      signal  sc_ff_set       :   std_logic;
     66  
     67      -- Global enable signal
     68      -- ====================
     69      signal  global_en       :   std_logic;
     70      signal  global_clr      :   std_logic;
     71  
     72  
     73      -- Change detector signals
     74      -- =======================
     75      signal  chg_det_reg     :   std_logic_vector(SIG_W-1 downto 0);
     76      signal  chg_det_en      :   std_logic;
     77      signal  chg_det_out     :   std_logic;
     78  
     79  begin
     80      ---------------------------------------------------------------------
     81      L_GLOBAL:   block
     82                  begin
     83  
     84                      -- Processor or a component can enable the module
     85                      global_en   <=  wdata_reg(0) or  trig_in_sc_ff;
     86                      global_clr  <=  wdata_reg(1);
     87  
     88                      sc_ff_set   <=  trig_in;
     89  
     90                      -- Counter is incremented when the module is globally enabled
     91                      -- and there was a change 'signal_from_DUV' vector
     92                      cntr_en_w   <=  global_en   and     chg_det_out;
     93  
     94                  end block;
     95      ---------------------------------------------------------------------
     96      L_RDATA:    block
     97                  begin
     98                      rdata   <= cntr_out_w;
     99                  end block;
     100      ---------------------------------------------------------------------
     101      L_WDATA:    block
     102                  begin
     103  
     104                      process(clk, rstn) is
     105                      begin
     106                          if(rstn = '0')  then
     107                              wdata_reg   <= (others =>'0');
     108                          elsif(rising_edge(clk)) then
     109  
     110                              if(wr = '1')    then
     111                                  wdata_reg <= wdata;
     112                              end if;
     113  
     114                          end if;
     115                      end process;
     116  
     117                  end block;
     118      ---------------------------------------------------------------------
     119      L_REG_LAYER:    block
     120                      begin
     121                          ------------------------------------------------
     122                          L_REGLAY:   if(REG_LAYER = true)    generate
     123  
     124                                          process(clk,rstn) is
     125                                          begin
     126                                              if(rstn = '0')  then
     127                                                  trig_out_reg    <= '0';
     128                                              elsif(rising_edge(clk)) then
     129  
     130                                                  -- Drive the 'chg_det_out' signal from flop
     131                                                  if(trig_out_reg_en = '1') then
     132                                                      trig_out_reg    <= chg_det_out;
     133                                                  end if;
     134  
     135                                              end if;
     136                                          end process;
     137  
     138                                          -- 'trig_out_reg' is enabled globally
     139                                          trig_out_reg_en <= global_en;
     140  
     141                                          -- Drive the output from the extra register
     142                                          trig_out    <= trig_out_reg;
     143  
     144                                      end generate;
     145                          ------------------------------------------------
     146                          L_NO_REGL:  if(REG_LAYER = false)   generate
     147  
     148                                          -- Drive the output directly
     149                                          trig_out    <= chg_det_out;
     150                                      end generate;
     151                          ------------------------------------------------
     152                      end block;
     153      ---------------------------------------------------------------------
     154      L_CHG_DET:  block
     155                  begin
     156                      chg_det_en  <=  global_en;
     157  
     158                      chg_det_out <=  '1' when( chg_det_reg /= signal_from_DUV
     159                                              ) else '0';
     160  
     161                      process(clk,rstn)   is
     162                      begin
     163                          if(rstn = '0')  then
     164                              chg_det_reg <= (others => '0');
     165  
     166                          elsif(rising_edge(clk)) then
     167  
     168                              if(chg_det_en = '1')    then
     169                                  chg_det_reg <= signal_from_DUV;
     170                              end if;
     171  
     172                          end if;
     173                      end process;
     174                  end block;
     175      ---------------------------------------------------------------------
     176      L_SC_FF:    entity work.sc_ff(rtl)
     177                      port map(
     178                          clk         => clk          ,
     179                          rstn        => rstn         ,
     180                          set         => sc_ff_set    ,
     181                          clr         => global_clr   ,
     182                          q           => trig_in_sc_ff
     183                      );
     184      ---------------------------------------------------------------------
     185      L_CNTR:     entity work.cntr(rtl)
     186                      generic map(
     187                          CW       =>  CW
     188                      )
     189                      port map(
     190                          clk         => clk          ,
     191                          rstn        => rstn         ,
     192                          en          => cntr_en_w    ,
     193                          clr         => global_clr   ,
     194                          cntr_out    => cntr_out_w
     195                      );
     196      ---------------------------------------------------------------------
     197  end architecture rtl;
